DECLARE
    v_cursor SYS_REFCURSOR;
    -- Definimos el registro con tipos de datos más flexibles para la prueba
    TYPE t_registro IS RECORD (
        nu_cui                 VARCHAR2(4000),
        de_pre_nombres         VARCHAR2(4000),
        de_primer_apellido     VARCHAR2(4000),
        de_segundo_apellido    VARCHAR2(4000),
        fe_nacimiento          VARCHAR2(4000), -- Cambiado a VARCHAR para evitar ORA-01861
        vinculo_prog1          VARCHAR2(4000),
        tipo_doc_prog1         VARCHAR2(4000),
        num_doc_prog1          VARCHAR2(4000),
        vinculo_prog2          VARCHAR2(4000),
        tipo_doc_prog2         VARCHAR2(4000),
        num_doc_prog2          VARCHAR2(4000),
        imagen_anverso         BLOB,
        imagen_reverso         BLOB
    );
    r_datos t_registro;
    v_encontrado BOOLEAN := FALSE;
BEGIN
    -- 1. Ejecución del procedimiento
    SP_ACTA_NACIMIENTO(1005583752, v_cursor);

    -- 2. Extracción de datos
    LOOP
        FETCH v_cursor INTO r_datos;
        EXIT WHEN v_cursor%NOTFOUND;

        v_encontrado := TRUE;
        DBMS_OUTPUT.PUT_LINE('--- DATOS ENCONTRADOS ---');
        DBMS_OUTPUT.PUT_LINE('CUI: ' || r_datos.nu_cui);
        DBMS_OUTPUT.PUT_LINE('Nombre: ' || r_datos.de_pre_nombres);
        DBMS_OUTPUT.PUT_LINE('Fecha Nac: ' || r_datos.fe_nacimiento);
    END LOOP;

    IF NOT v_encontrado THEN
        DBMS_OUTPUT.PUT_LINE('No se encontró el acta especificada.');
    END IF;

    CLOSE v_cursor;
EXCEPTION
    WHEN OTHERS THEN
        IF v_cursor%ISOPEN THEN CLOSE v_cursor; END IF;
        DBMS_OUTPUT.PUT_LINE('Error: ' || SQLERRM);
END;
